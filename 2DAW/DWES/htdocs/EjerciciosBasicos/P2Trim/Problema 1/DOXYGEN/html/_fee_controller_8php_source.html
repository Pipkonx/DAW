146 public function update(Request $request, Fee $fee): RedirectResponse
147 {
148 $validated = $request->validate([
149 'concept' => 'required|string|max:255',
150 'emission_date' => 'required|date',
151 'amount' => 'required|numeric|min:0',
152 'is_paid' => 'required|in:S,N',
153 'payment_date' => 'nullable|date|required_if:is_paid,S',
154 'notes' => 'nullable|string',
155 ]);
156
157 // Si se marca como pagada y no lo estaba, usamos la lógica del modelo
158 if ($validated['is_paid'] === 'S' && $fee->is_paid !== 'S') {
159 $fee->markAsPaid();
160 // Actualizamos solo los campos que NO gestiona markAsPaid()
161 unset($validated['is_paid'], $validated['payment_date']);
162 }
163
164 $fee->update($validated);
165
166 return redirect()->route('fees.index')->with('success', 'Cuota actualizada.');
